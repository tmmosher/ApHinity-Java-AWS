create table public.app_user
(
    id                bigserial
        primary key,
    email             text                                   not null
        unique,
    email_verified_at timestamp with time zone,
    password_hash     text,
    created_at        timestamp with time zone default now() not null,
    updated_at        timestamp with time zone default now() not null,
    name              varchar(255)
);

alter table public.app_user
    owner to postgres;

create table public.role
(
    id   bigserial
        primary key,
    name text not null
        unique
);

alter table public.role
    owner to postgres;

create table public.user_role
(
    user_id bigint not null

            on delete cascade,
    role_id bigint not null
        references public.role
            on delete cascade,
    primary key (user_id, role_id)
);

alter table public.user_role
    owner to postgres;

create table public.auth_session
(

create table public.location_invite
(
    id               bigserial
        primary key,
    location_id      bigint                                           not null
        references public.location
            on delete cascade,
    invited_email    text                                             not null,
    invited_by       bigint
                                                                      references public.app_user
                                                                          on delete set null,
    token_hash       text                                             not null
        unique,
    expires_at       timestamp with time zone                         not null,

    created_at       timestamp with time zone default now()           not null,
    accepted_at      timestamp with time zone,
    accepted_user_id bigint
                                                                      references public.app_user
                                                                          on delete set null,
    revoked_at       timestamp with time zone
);

alter table public.location_invite

create extension if not exists pg_trgm;

create table public.service_event
(
    id             bigserial
        primary key,
    location_id    bigint                                 not null
        references public.location
            on delete cascade,
    title          text                                   not null
        constraint service_event_title_not_blank_check
            check (btrim(title) <> ''::text),
    responsibility text                                   not null
        constraint service_event_responsibility_check
            check (responsibility = ANY (ARRAY ['client'::text, 'partner'::text])),
    event_date     date                                   not null,
    event_time     time without time zone                 not null,
    end_event_date date                                   not null,
    end_event_time time without time zone                 not null,
    description    text,
    status         text                     default 'upcoming'::text not null
        constraint service_event_status_check
            check (status = ANY (ARRAY ['upcoming'::text, 'current'::text, 'overdue'::text, 'completed'::text])),
    constraint service_event_range_check
        check ((end_event_date > event_date)
            or (end_event_date = event_date and end_event_time >= event_time)),
    created_at     timestamp with time zone default now() not null,
    updated_at     timestamp with time zone default now() not null
);

alter table public.service_event
    owner to postgres;

create index service_event_location_schedule_idx
    on public.service_event (location_id, event_date, event_time, end_event_date, end_event_time);

create index service_event_location_status_idx
    on public.service_event (location_id, status);

create index service_event_location_responsibility_idx
    on public.service_event (location_id, responsibility);

create index service_event_title_trgm_idx
    on public.service_event using gin (title gin_trgm_ops);

create index service_event_description_trgm_idx
    on public.service_event using gin (description gin_trgm_ops);

create table public.gantt_task
(
    id          bigserial
        primary key,
    location_id bigint                                 not null
        references public.location
            on delete cascade,
    title       text                                   not null
        constraint gantt_task_title_not_blank_check
            check (btrim(title) <> ''::text),
    start_date  date                                   not null,
    end_date    date                                   not null,
    description text,
    created_at  timestamp with time zone default now() not null,
    updated_at  timestamp with time zone default now() not null,
    constraint gantt_task_title_trigram_check
        check (char_length(btrim(title)) >= 3),
    constraint gantt_task_range_check
        check (end_date >= start_date)
);

alter table public.gantt_task
    owner to postgres;

create index gantt_task_location_schedule_idx
    on public.gantt_task (location_id, start_date, end_date);

create index gantt_task_title_trgm_idx
    on public.gantt_task using gin (title gin_trgm_ops);

create index gantt_task_description_trgm_idx
    on public.gantt_task using gin (description gin_trgm_ops);
